public class Pessoa {
    private String nome;
    private int idade;
    private double altura;
    private double peso;
    public Pessoa(String nome, int idade, double altura, double peso) {
        this.nome = nome;
        this.idade = idade;
        this.altura = altura;
        this.peso = peso;
    }
    public String getNome() {
        return nome;
    }
    public int getIdade() {
        return idade;
    }
    public double getAltura() {
        return altura;
    }
    public double getPeso() {
        return peso;
    }
    public double calcularIMC() {
        return peso / (altura * altura);
    }
    public String categoriaIMC() {
        double imc = calcularIMC();
        if (imc < 18.5)
            return "Abaixo do peso";
        else if (imc < 25)
            return "Peso normal";
        else if (imc < 30)
            return "Sobrepeso";
        else if (imc < 35)
            return "Obesidade grau 1";
        else if (imc < 40)
            return "Obesidade grau 2";
        else
            return "Obesidade grau 3";
    }
}

import java.util.ArrayList;
public class PessoaController {
    private ArrayList<Pessoa> pessoas;
    public PessoaController() {
        pessoas = new ArrayList<>();
    }
    public void adicionarPessoa(String nome, int idade, double altura, double peso) {
        Pessoa p = new Pessoa(nome, idade, altura, peso);
        pessoas.add(p);
    }
    public int quantidadePessoas() {
        return pessoas.size();
    }
    public boolean limiteAtingido() {
        return pessoas.size() == 5;
    }
    public String gerarResultados() {
        String texto = "";
        for (Pessoa p : pessoas) {
            texto += "Nome: " + p.getNome() + "\n";
            texto += "IMC: " + String.format("%.2f", p.calcularIMC()) + "\n";
            texto += "Categoria: " + p.categoriaIMC() + "\n\n";
        }
        return texto;
    }
    public String gerarEstatisticas() {
        double soma = 0;
        Pessoa menor = pessoas.get(0);
        Pessoa maior = pessoas.get(0);
        for (Pessoa p : pessoas) {
            double imc = p.calcularIMC();
            soma += imc;
            if (imc < menor.calcularIMC())
                menor = p;
            if (imc > maior.calcularIMC())
                maior = p;
        }
        double media = soma / pessoas.size();
        String texto = "";
        texto += "Média IMC: " + String.format("%.2f", media) + "\n\n";
        texto += "Menor IMC: " + menor.getNome() + "\n";
        texto += "Maior IMC: " + maior.getNome();
        return texto;
    }
}

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.*;
import androidx.appcompat.app.AppCompatActivity;
public class MainActivity extends AppCompatActivity {
    private EditText nome;
    private EditText idade;
    private EditText altura;
    private EditText peso;
    private TextView contador;
    private Button cadastrar;
    private ImageView estatistica;
    private PessoaController controller;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        nome = findViewById(R.id.editTextNome);
        idade = findViewById(R.id.editTextIdade);
        altura = findViewById(R.id.editTextAltura);
        peso = findViewById(R.id.editTextPeso);
        contador = findViewById(R.id.textViewContador);
        cadastrar = findViewById(R.id.buttonCalcular);
        estatistica = findViewById(R.id.imageViewEstatistica);
        controller = new PessoaController();
        atualizarContador();
        cadastrar.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                cadastrarPessoa();
            }
        });
        estatistica.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                abrirEstatisticas();
            }
        });
    }
    private void cadastrarPessoa() {
        String n = nome.getText().toString();
        int i = Integer.parseInt(idade.getText().toString());
        double a = Double.parseDouble(altura.getText().toString());
        double p = Double.parseDouble(peso.getText().toString());
        controller.adicionarPessoa(n, i, a, p);
        atualizarContador();
        limparCampos();
        if (controller.limiteAtingido()) {
            abrirResultados();
        } else {
            Toast.makeText(MainActivity.this, "Pessoa cadastrada", Toast.LENGTH_SHORT).show();
        }
    }
    private void abrirResultados() {
        Intent intent = new Intent(MainActivity.this, ResultadoActivity.class);
        intent.putExtra("titulo","Resultados IMC");
        intent.putExtra("resultado", controller.gerarResultados());
        startActivity(intent);
    }
    private void abrirEstatisticas() {
        if (!controller.limiteAtingido()) {
            Toast.makeText(MainActivity.this,"Cadastre 5 pessoas",Toast.LENGTH_SHORT).show();
            return;
        }
        Intent intent = new Intent(MainActivity.this, ResultadoActivity.class);
        intent.putExtra("titulo","Estatísticas");
        intent.putExtra("resultado", controller.gerarEstatisticas());
        startActivity(intent);
    }
    private void limparCampos() {
        nome.setText("");
        idade.setText("");
        altura.setText("");
        peso.setText("");
    }
    private void atualizarContador() {
        contador.setText("Pessoas cadastradas: " +
                controller.quantidadePessoas() + "/5");
    }
}

import com.example.correcaocalculadora.model.CalculadoraModel;
public class CalculadoraController {
    private CalculadoraModel model;
    public CalculadoraController() {
        model = new CalculadoraModel();
    }
    public String calcular(String texto1, String texto2, String operacao) {
        if (texto1.isEmpty()) {
            return "Digite o primeiro valor";
        }
        if (texto2.isEmpty()) {
            return "Digite o segundo valor";
        }
        double valor1 = Double.parseDouble(texto1);
        double valor2 = Double.parseDouble(texto2);
        double resultado;
        String texto;
        if (operacao.equals("+")) {
            resultado = model.somar(valor1, valor2);
            texto = valor1 + " + " + valor2 + " = " + resultado;
        } else if (operacao.equals("-")) {
            resultado = model.subtrair(valor1, valor2);
            texto = valor1 + " - " + valor2 + " = " + resultado;
        } else if (operacao.equals("*")) {
            resultado = model.multiplicar(valor1, valor2);
            texto = valor1 + " * " + valor2 + " = " + resultado;
        } else if (operacao.equals("/")) {
            if (valor2 == 0) {
                return "Não pode dividir por zero";
            }
            resultado = model.dividir(valor1, valor2);
            texto = valor1 + " / " + valor2 + " = " + resultado;
        } else {
            return "Operação inválida";
        }
        return texto;
    }
}

public class Calculo {
    private static final double PRECO_CAFE = 2.50;
    private static final double PRECO_BOLO = 5.00;

    public Calculo() {
    }
    public double adicionarCafe(int qtd) {
        if (qtd > 0) {
            return qtd * PRECO_CAFE;
        }
        return 0;
    }
    public double adicionarBolo(int qtd) {
        if (qtd > 0) {
            return qtd * PRECO_BOLO;
        }
        return 0;
    }
}//class

public class Controller {
    private  Calculo model;
    public Controller() {
        this.model = new Calculo();
    }
    public Double adicionar(String texto, String textoQuantidade) {
        int qtd = Integer.parseInt(textoQuantidade);
        if(texto.equals("Café")){
            return model.adicionarCafe(qtd);
        }
        if(texto.equals("Bolo")){
            return model.adicionarBolo(qtd);
        }
        return 0.0;
    }
}//class

public class CafeActivity extends AppCompatActivity {
    private EditText editTextQuantidade;
    private TextView textViewCafe;
    private Button buttonAdd;
    private Controller controller;
    private Double resultado = 0.0;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_cafe);
        controller = new Controller();

        if(intent != null && intent.hasExtra("total")){
